## Model

```javascript
function rollDice(numDice, numSides) {
    let roll = [];
    for (let i = 0; i < numDice; i++) {
        roll.push(floor(random(1, numSides + 1)));
    }
    return roll;
}

function calculateCapacity(d, len) {
    return 3.14 * (d / 2) * (d / 2) * len;
}

function calculateMaterialsCost(d, len) {
    return 3.14 * d * len;
}
```

## View

```javascript
function setupView() {
    createCanvas(800, 600);
    background(200, 200, 80);

    textSize(20);
    fill(0);
    noStroke();

    text("Press r to roll the dice", 50, 100);
    text("Press p to pipeline cost", 50, 140);
}

function displayRoll(roll) {
    print("You rolled: " + roll);
}

function displayCapacity(volume) {
    print("Total capacity: " + volume + " cubic meters.");
}

function displayMaterialsCost(squareMeters) {
    print("Sheet metal costs $50 per square meter");
    print("Total metal needed: " + squareMeters);
    print("Total Cost: $" + (squareMeters * 50));
}
```

## Controller


```javascript
function keyPressedController() {
    if (key === 'r') {
        handleRollDice(8, 20);
    }
    if (key === 'p') {
        handlePipeline();
    }
}

function handleRollDice(numDice, numSides) {
    let roll = rollDice(numDice, numSides);
    displayRoll(roll);
}

function handlePipeline() {
    let diameter = parseFloat(window.prompt("What is the diameter of the pipeline in meters?"));
    let length = parseFloat(window.prompt("What is the length in kilometers?"));

    length *= 1000; // Convert km to meters

    let squareMeters = calculateMaterialsCost(diameter, length);
    let volume = calculateCapacity(diameter, length);

    displayMaterialsCost(squareMeters);
    displayCapacity(volume);
}
```
