```javascript
let allBands = ["The Beatles", "BTS", "Lil Nas X", "ACDC", "The Weeknd", "Migos", "Juice WRLD", "Halsey", "The Kid LAROI"];

let currentMessage = "";
let score = 0;

function addBand(bandName) {
  allBands.push(bandName);

function searchBand(bandName) {
  return allBands.includes(bandName);
}
```

## Controller
The Controller will use functions to handle user input and update the Model and View.

```javascript
function keyPressed() {
  if (key === 'b') {
    currentBand = prompt("What is your proposed band name?");
    if (searchBand(currentBand)) {
      currentMessage = `Sorry, this name already exists.`;
      if (currentBand.includes("s")) {
        currentMessage += ` Perhaps try ${currentBand.replace("s", "$$$")}.`;
      }
    } else {
      currentMessage = `Congratulations. This is a new name. Press 'a' to analyze.`;
      addBand(currentBand);
    }
    displayMessage(currentMessage);
  } else if (key === 'a' && currentBand) {
    score = analyzeBandName(currentBand);
    displayScore(currentBand, score);
  }
}

function analyzeBandName(bandName) {
  let score = 0;
  if (bandName.length > 5) score += 1;
  if (/[A-Z]/.test(bandName)) score += 5;
  if (/[0-9]/.test(bandName)) score += 1;
  if (/[!@#$%^&*(),.?":{}|<>]/.test(bandName)) score += 10;
  if (bandName.toLowerCase().includes("band")) score += 100;
  return score;
}
```
